# 利用双重循环计算累计人数

  for( j in seq(2,areas)){              #每个区都计算
    clear_yq[i,j] <- clear_yq[i-1,j]+clear_yq[i,j]

# 计算列数
col.search <- dim(all_area_data)[2]
# 生成一个表示疫情发展第n天的序列
during_seq <- seq(1,during)
# 生成一个向量来“盛放”每天全上海的累计确诊人数
sum_areas <- rep(0,time=during)

利用循环来计算每天全上海的累计确诊人数

for (i in seq(1,during)){
    sum_areas[i] = all_area_data[2*i-1,col.search]+all_area_data[2*i,col.search]
}
sum_areas <- rev(sum_areas) #把从6.31-3.1排列改成3.1-5.31

画图并拟合

  • 第一列表示疫情发展第n天；第二列表示该天全城累计确诊病例
  • 拐点定义：拐点，又称反曲点，在数学上指改变曲线向上或向下方向的点，直观地说拐点是使切线穿越曲线的点（即连续曲线的凹弧与凸弧的分界点）。若该曲线图形的函数在拐点有二阶导数，则二阶导数在拐点处异号（由正变负或由负变正）或不存在。
  • 在本情境中，疫情拐点指的是全城累计确诊人数的二阶导数为0的点，也即每日变化的累计人数的变化值的方向变化的时候
final.sum <- data.frame(during_seq,sum_areas)

plot(final.sum$during_seq,final.sum$sum_areas,type="l") #画图展现走势

#局部回归分析的非参数方法，它主要是把样本划分成一个个小区间，对区间中的样本进行多项式拟合，不断重复这个过程得到在不同区间的加权回归曲线，最后再把这些回归曲线的中心连在一起合成完整的回归曲线
lo <- loess(final.sum$sum_areas~final.sum$during_seq)
xl <- seq(min(final.sum$during_seq),max(final.sum$during_seq), (max(final.sum$during_seq) - min(final.sum$during_seq))/during)
out = predict(lo,xl)
lines(xl, out, col='red', lwd=2)
#找到拐点
infl <- c(FALSE, diff(diff(out)>0)!=0) #此代码找到所有“转折点”，无论它向上还是向下
points(xl[infl ], out[infl ], col="blue")
